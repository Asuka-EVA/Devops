```go
package main
import (
"fmt"
"strconv"
)
func main() {
num := 100
strNum := strconv.Itoa(num)
fmt.Printf("num: %T %v \n", num, num)
fmt.Printf("strNum: %T %v \n", strNum, strNum)
intNum, _ := strconv.Atoi(strNum)
fmt.Printf("intNum: %T %v \n", intNum, intNum)
}
/*
num: int 100
strNum: string 100
intNum: int 100
*/
```
