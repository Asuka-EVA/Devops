1.关闭SELinux

```bash
sed -i ‘s/SELINUX=enforcing/SELINUX=disabled/’ /etc/selinux/config

grep SELINUX=disabled /etc/selinux/config

setenforce 0

getenforce
```

2. 关闭iptables

```bash
/etc/init.d/iptables stop

/etc/init.d/iptables stop

chkconfig iptables off
```

3. 精简开机自启动服务

```bash
chkconfig|egrep -v "crond|sshd|network|rsyslog|sysstat"|awk '{print "chkconfig",$1,"off"}'|bash

export LANG=en

chkconfig --list|grep 3:on
```

4. 提权jack可以sudo

```bash
useradd jack

\cp /etc/sudoers /etc/sudoers.ori

echo “jack ALL=(ALL) NOPASSWD:ALL ” >>/etc/sudoers

tail -l /etc/sudoers

visudo -c
```

5. 中文字符集

```bash
cp /etc/sysconfig/i18n /etc/sysconfig/i18n.ori

echo 'LANG="zh_CN.UTF-8"' >/etc/sysconfig/i18n

source /etc/sysconfig/i18n

echo $LANG
```

6. 时间同步

```bash
echo '#time syne by jack at 2010-2-1' >>/car/spool/cron/root

echo '/5 * * * * /user/sbin/ntpdate time.nist.gov >/dev/null 2>&1' >>/var/spool/cron/root

crontab -l
```

7. 命令行安全

```bash
echo ‘export TMOUT=300’ >>/etc/profile

echo ‘export HISTSIZE=5’ >>/etc/profile

echo ‘export HSTFILESIZE=5’ >>/etc/profile

tail -3 /etc/profile

. /etc/profile
```

8. 加大文件描述符

```bash
echo '* - nofile 65535'>>/etc/security/limits.conf

tail -1 /etc/security/limits.conf
```

9. 内核优化

```bash
cat >>/etc/sysctl.conf<<EOF

net.ipv4.tcp_fin_timeout = 2

net.ipv4.tcp_tw_reuse = 1

net.ipv4.tcp_tw_recycle = 1

net.ipv4.tcp_syncookies = 1

net.ipv4.tcp_keepalive_time = 600

net.ipv4.ip_local_port_range = 400065000

net.ipv4.tcp_max_syn_backlog = 16384

net.ipv4.tcp_max_tw_buckets = 36000

net.ipv4.route.gc_timeout = 100

net.ipv4.tcp_syn_retries = 1

net.ipv4.tcp_synack_retries = 1

net.core.somaxconn = 16384

net.core.netdev_max_backlog = 16384

net.ipv4.tcp_max_orphans = 16384

\#以下参数是对iptables防火墙的优化，防火墙不开会提示，可以忽略不理。

net.ipv4.ip_conntrack_max = 25000000

net.ipv4.netfilter.ip_conntrack_max=25000000

net.ipv4.netfilter.ip_conntrack_tcp_timeout_established=180

net.ipv4.netfilter.ip_conntrack_tcp_timeout_time_wait=120

net.ipv4.netfilter.ip_conntrack_tcp_timeout_close_wait=60

net.ipv4.netfilter.ip_conntrack_tcp_timeout_fin_wait=120

EOF
```
